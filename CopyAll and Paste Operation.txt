#include <stdio.h>
#include <string.h>

int main() {
    int num;
    scanf("%d", &num);
    getchar(); // Consume the newline character

    char sen[101];
    fgets(sen, 101, stdin);
    sen[strcspn(sen, "\n")] = '\0'; // Remove the trailing newline character

    long sum = strlen(sen);

    for (int i = 1; i <= num; i++) {
        sum += sum;
    }

    printf("%ld", sum);
    return 0;
}
