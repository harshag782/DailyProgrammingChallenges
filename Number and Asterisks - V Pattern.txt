#include <stdio.h>

int main() {
    int N;
    printf("Enter the value of N: ");
    scanf("%d", &N);

    // Print the pattern
    for (int i = 1; i <= N; i++) {
        // Print leading asterisks
        for (int j = 1; j < i; j++) {
            printf("*");
        }

        // Print numbers
        for (int k = i; k <= N; k++) {
            printf("%d", k);

            // Print asterisks between numbers
            if (k < N) {
                printf("*");
            }
        }

        // Print trailing asterisks
        for (int j = 1; j < i; j++) {
            printf("*");
        }

        printf("\n");
    }

    return 0;
}
